cmake_minimum_required(VERSION 2.8.7)

set(DELEGATE_SRC_DIR ${CMAKE_CURRENT_SOURCE_DIR})
IF(WITH_GPU)
    set(GPU_DELEGATE_SRC
        ${DELEGATE_SRC_DIR}/GpuDelegate.cpp
    )
ENDIF(WITH_GPU)

IF(WITH_ARMNN)
    set(ARMNN_DELEGATE_SRC
        ${DELEGATE_SRC_DIR}/ArmNNDelegate.cpp
    )
ENDIF(WITH_ARMNN)

IF(WITH_NPU)
    set(NPU_DELEGATE_SRC
        ${DELEGATE_SRC_DIR}/NpuDelegate.cpp
        ${DELEGATE_SRC_DIR}/NnapiDelegate.cpp
    )
ENDIF(WITH_NPU)

IF(WITH_NNAPI)
    set(NNAPI_DELEGATE_SRC
        ${DELEGATE_SRC_DIR}/NnapiDelegate.cpp
    )
ENDIF(WITH_NNAPI)

IF(WITH_XNNPACK)
    set(XNNPACK_DELEGATE_SRC
    )
ENDIF(WITH_XNNPACK)

target_include_directories(${PROJECT_NAME} PRIVATE ${DELEGATE_SRC_DIR})
target_sources(${PROJECT_NAME} PRIVATE
    ${GPU_DELEGATE_SRC} ${ARMNN_DELEGATE_SRC} ${XNNPACK_DELEGATE_SRC} ${NPU_DELEGATE_SRC} ${NNAPI_DELEGATE_SRC})
